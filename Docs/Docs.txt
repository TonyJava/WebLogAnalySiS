CREATE TABLE tgt_weblog (
dateformat String,
keywordsString String,
url String
)
PARTITIONED BY(year INT, month INT)
ROW FORMAT DELIMITED FIELDS TERMINATED BY ','
STORED AS TEXTFILE;



INSERT OVERWRITE TABLE tgt_weblog
PARTITION (year,month)

from stg_weblog
